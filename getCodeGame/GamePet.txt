create table dbo.GamePet (
   ActorID              int                  not null,
   EmployerActorID      int                  null,
   PetID                int                  identity(1,1),
   PetName              varchar(19)          not null,
   Sex                  tinyint              null,
   Ability              smallint             null,
   PetType              tinyint              null,
   Level                tinyint              not null constraint DF_GamePet_PetLevel default (0),
   Exp                  int                  not null constraint DF_GamePet_PetExpValue default (0),
   CurHP                int                  not null constraint DF_GamePet_CurHP default (0),
   Life                 int                  null,
   IsReadyFight         tinyint              null,
   CurThew              int                  null,
   WiseLevel            tinyint              null,
   WiseSoulLevel        tinyint              null,
   WiseBlessLevel       tinyint              null,
   WiseBless            int                  null,
   GrowLevel            tinyint              null,
   GrowSoulLevel        tinyint              null,
   GrowBlessLevel       tinyint              null,
   GrowBless            int                  null,
   KitbagGridNum        tinyint              null,
   TransformID          int                  null,
   SkillBuffer          binary(24)           not null constraint DF_GamePet_SkillBuffer default 0x00,
   CreateTime           datetime             not null constraint DF_GamePet_CreateTime default getdate(),
   IsDelete             bit                  not null constraint DF_GamePet_IsDelete default (0)
)
